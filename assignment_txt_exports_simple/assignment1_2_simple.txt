log_freqs = np.log10(sorted_freqs)
slope, intercept = np.polyfit(log_ranks, log_freqs, 1)
fit_line = 10 ** (intercept + slope * log_ranks)

plt.title("Zipf's Law in Marathi Corpus")

  NP: {<DT>?<JJ>*<NN.*>+}  # Noun phrases

        else:
            row.append(0)
    matrix.append(row)

def get_rhyme_part(phonemes):
    """Extract the part of pronunciation used for rhyming"""
    for i, phoneme in enumerate(phonemes):
        if phoneme[-1].isdigit():  # Find first stressed vowel
            return phonemes[i:]
    return phonemes
