from collections import Counter
from itertools import chain

import matplotlib.pyplot as plt
import numpy as np
import seaborn as sns
from nltk import FreqDist
from nltk.corpus import indian

sns.set_style("whitegrid")

# Directly use Nirmala UI for Devanagari rendering
plt.rcParams["font.family"] = "Nirmala UI"
plt.rcParams["axes.unicode_minus"] = False

file_ids = [fid for fid in indian.fileids() if "marathi" in fid.lower()]
if not file_ids:
    raise ValueError("No Marathi dataset found in nltk.corpus.indian.")

print("Using Indic font: Nirmala UI")
print("Marathi files used:", file_ids)

def is_valid_word(token):
    return isinstance(token, str) and any(ch.isalpha() for ch in token)

# Build corpus-level token list
all_words_raw = list(chain.from_iterable(indian.words(fid) for fid in file_ids))
all_words = [w.lower() for w in all_words_raw if is_valid_word(w)]

if not all_words:
    raise ValueError("No alphabetic words found in Marathi dataset.")

# 1) Top frequent words
top_words = FreqDist(all_words).most_common(20)
words, word_counts = zip(*top_words)

plt.figure(figsize=(12, 5))
sns.barplot(x=list(words), y=list(word_counts), color="#4C72B0")
plt.title("Top 20 Most Frequent Words in Marathi Corpus")
plt.xlabel("Word")
plt.ylabel("Frequency")
plt.xticks(rotation=45, ha="right")
plt.tight_layout()
plt.show()

# 2) Word-length distribution
word_lengths = [len(w) for w in all_words]

plt.figure(figsize=(10, 5))
sns.histplot(word_lengths, bins=range(1, max(word_lengths) + 2), color="teal", kde=False)
plt.title("Word Length Distribution in Marathi Corpus")
plt.xlabel("Word Length")
plt.ylabel("Count")
plt.tight_layout()
plt.show()

# 3) POS tag distribution
all_tagged = list(chain.from_iterable(indian.tagged_words(fid) for fid in file_ids))
pos_tags = [tag for _, tag in all_tagged if tag]
top_pos = Counter(pos_tags).most_common(15)
pos_labels, pos_counts = zip(*top_pos)

plt.figure(figsize=(12, 5))
sns.barplot(x=list(pos_labels), y=list(pos_counts), color="#DD8452")
plt.title("Top 15 POS Tags in Marathi Corpus")
plt.xlabel("POS Tag")
plt.ylabel("Frequency")
plt.xticks(rotation=45, ha="right")
plt.tight_layout()
plt.show()

# 4) Top bigrams (bounded for speed)
max_tokens_for_bigrams = 200000
bigram_words = all_words[:max_tokens_for_bigrams]
bigram_freq = Counter(zip(bigram_words, bigram_words[1:])).most_common(15)
bigram_labels = [f"{w1} {w2}" for (w1, w2), _ in bigram_freq]
bigram_counts = [c for _, c in bigram_freq]

plt.figure(figsize=(14, 5))
sns.barplot(x=bigram_labels, y=bigram_counts, color="#55A868")
plt.title("Top 15 Bigrams in Marathi Corpus")

plt.xticks(rotation=50, ha="right")
plt.tight_layout()
plt.show()

# 5) Zipf's law in Marathi corpus (rank-frequency on log-log scale)
freq_dist = FreqDist(all_words)
sorted_freqs = np.array(sorted(freq_dist.values(), reverse=True))
ranks = np.arange(1, len(sorted_freqs) + 1)

# Fit line in log-log space for comparison
log_ranks = np.log10(ranks)
log_freqs = np.log10(sorted_freqs)
slope, intercept = np.polyfit(log_ranks, log_freqs, 1)
fit_line = 10 ** (intercept + slope * log_ranks)

plt.figure(figsize=(12, 6))
plt.loglog(ranks, sorted_freqs, marker=".", linestyle="none", alpha=0.6, label="Observed word frequencies")
plt.loglog(ranks, fit_line, color="#C44E52", linewidth=2, label=f"Fitted trend (slope={slope:.2f})")
plt.title("Zipf's Law in Marathi Corpus")
plt.xlabel("Word Rank (log scale)")
plt.ylabel("Frequency (log scale)")
plt.legend()
plt.tight_layout()
plt.show()

from nltk import FreqDist, ngrams, collocations

print("\n" + "=" * 60)
print("3. COLLOCATIONS - Meaningful Word Pairs")

# Use bigram collocation finder
brown_bigram_measures = collocations.BigramAssocMeasures()

reuters_bigram_measures = collocations.BigramAssocMeasures()

chunk_grammar = r"""
  NP: {<DT>?<JJ>*<NN.*>+}  # Noun phrases
  VP: {<VB.*><NP|PP>}       # Verb phrases

    print(f"{'='*60}")
    
    synsets = wn.synsets(word)
    print(f"\nNumber of synsets: {len(synsets)}")
    
    for i, syn in enumerate(synsets[:3], 1):
        print(f"\n{i}. Synset: {syn.name()}")
        print(f"   POS: {syn.pos()} (n=noun, v=verb, a=adjective, r=adverb)")
        print(f"   Definition: {syn.definition()}")
        print(f"   Examples: {syn.examples()}")
        print(f"   Lemmas: {[lemma.name() for lemma in syn.lemmas()]}")
    
    if synsets:
        first_synset = synsets[0]
        print(f"\nSynonyms of '{word}' (from first synset):")
        synonyms = {lemma.name() for lemma in first_synset.lemmas()}
        print(f"   {synonyms}")
        
        print(f"\nAntonyms of '{word}' (if available):")
        antonyms = set()
        for lemma in first_synset.lemmas():
            if lemma.antonyms():
                antonyms.add(lemma.antonyms()[0].name())
        print(f"   {antonyms if antonyms else 'No direct antonyms found'}")

]

# ngrams with probabilities
def get_ngram_probabilities(data, n=2):
    all_tokens = [word for sublist in data for word in sublist]
    # generate ngrams
    n_grams = list(ngrams(all_tokens, n))
    n_minus_1_grams = list(ngrams(all_tokens, n - 1))

    counts = Counter(n_grams)
    history_counts = Counter(n_minus_1_grams)

    print(f"\nTop {n}-gram Probabilities:")
    for gram in list(counts.keys())[:5]:
        # P(word | history) = count(history + word) / count(history)
        history = gram[:-1]
        prob = counts[gram] / history_counts[history]
        print(f"P({gram[-1]} | {' '.join(history)}) = {prob:.4f}")

def build_next_word_probability_model(data, n=2):
    all_tokens = [word for sublist in data for word in sublist]
    n_grams = list(ngrams(all_tokens, n))
    n_minus_1_grams = list(ngrams(all_tokens, n - 1))

    counts = Counter(n_grams)
    history_counts = Counter(n_minus_1_grams)

    prob_model = {}
    for gram, count in counts.items():
        history = gram[:-1]
        next_word = gram[-1]
        prob = count / history_counts[history]
        if history not in prob_model:
            prob_model[history] = {}
        prob_model[history][next_word] = prob

    return prob_model, counts

# Predict next word given context using probability model
def predict_next_word(context, prob_model):
    context_tuple = tuple(context.lower().split())

    if context_tuple not in prob_model:
        return None, None, "Context not found in model"

    next_words = prob_model[context_tuple]
    predicted_word = max(next_words, key=next_words.get)
    probability = next_words[predicted_word]

    return predicted_word, probability, None

from nltk.tree import Tree

def ieer_tree_to_tokens_iob(node):
    tokens_out = []
    tags_out = []

    for child in node:
        if isinstance(child, Tree):
            label = child.label()
            child_tokens = [str(tok) for tok in child.leaves()]
            for i, tok in enumerate(child_tokens):
                prefix = "B" if i == 0 else "I"
                tokens_out.append(tok)
                tags_out.append(f"{prefix}-{label}")
        else:
            tokens_out.append(str(child))
            tags_out.append("O")

    return tokens_out, tags_out

def translate_batch(model_family, texts, src_key, tgt_key, max_new_tokens=128):
    """Translate a list of texts using selected model family."""
    bundle = models[model_family]
    tokenizer = bundle["tokenizer"]
    model = bundle["model"]

    if model_family == "NLLB":
        src_lang = NLLB_LANGS[src_key]
        tgt_lang = NLLB_LANGS[tgt_key]
        tokenizer.src_lang = src_lang
        encoded = tokenizer(texts, return_tensors="pt", padding=True, truncation=True).to(device)
        forced_bos = tokenizer.convert_tokens_to_ids(tgt_lang)
        with torch.no_grad():
            generated = model.generate(**encoded, forced_bos_token_id=forced_bos, max_new_tokens=max_new_tokens)
    else:  # M2M100
        src_lang = M2M_LANGS[src_key]
        tgt_lang = M2M_LANGS[tgt_key]
        tokenizer.src_lang = src_lang
        encoded = tokenizer(texts, return_tensors="pt", padding=True, truncation=True).to(device)
        forced_bos = tokenizer.get_lang_id(tgt_lang)
        with torch.no_grad():
            generated = model.generate(**encoded, forced_bos_token_id=forced_bos, max_new_tokens=max_new_tokens)

    return tokenizer.batch_decode(generated, skip_special_tokens=True)


def evaluate_set(df, data_type, language_pairs):
    bleu_metric = evaluate.load("sacrebleu")
    chrf_metric = evaluate.load("chrf")
    ter_metric = evaluate.load("ter")

    rows = []
    for src, tgt in language_pairs:
        src_texts = df[src].astype(str).tolist()
        ref_texts = df[tgt].astype(str).tolist()
        wrapped_refs = [[ref] for ref in ref_texts]

        for model_family in models.keys():
            hyps = translate_batch(
                model_family=model_family,
                texts=src_texts,
                src_key=src,
                tgt_key=tgt,
            )

            bleu_score = bleu_metric.compute(predictions=hyps, references=wrapped_refs)["score"]
            chrf_score = chrf_metric.compute(predictions=hyps, references=wrapped_refs)["score"]
            ter_score = ter_metric.compute(predictions=hyps, references=wrapped_refs)["score"]

            rows.append(
                {
                    "data_type": data_type,
                    "model_family": model_family,
                    "pair": f"{src}->{tgt}",
                    "BLEU": bleu_score,
                    "chrF": chrf_score,
                    "TER": ter_score,
                }
            )

    return pd.DataFrame(rows)

        print(f"Loaded model: {key} -> {model_name}")
    return models


def compound_to_label(score):
    if score >= 0.05:
        return "POSITIVE"
    if score <= -0.05:
        return "NEGATIVE"
    return "NEUTRAL"


def lexicon_rule_predict(text, lexicon_model):
    words = tokenize_words(text)
    pos_hits = sum(1 for w in words if w in lexicon_model["pos_words"])
    neg_hits = sum(1 for w in words if w in lexicon_model["neg_words"])
    total_hits = pos_hits + neg_hits

    if total_hits == 0:
        return "NEUTRAL", 0.5

    raw_score = (pos_hits - neg_hits) / total_hits
    label = compound_to_label(raw_score)
    confidence = min(0.99, 0.5 + abs(raw_score) / 2)
    return label, confidence


def analyze_sentiment(text, sentiment_models):
    """Run both models and return model-wise outputs with a final voting label."""
    model_results = {}
    votes = {"POSITIVE": 0, "NEGATIVE": 0, "NEUTRAL": 0}

    vader_scores = sentiment_models["vader"].polarity_scores(text)
    vader_label = compound_to_label(vader_scores["compound"])
    vader_conf = abs(float(vader_scores["compound"]))
    model_results["vader"] = {
        "raw_label": vader_label,
        "mapped_label": vader_label,
        "score": vader_conf,
    }
    votes[vader_label] += 1

    lex_label, lex_score = lexicon_rule_predict(text, sentiment_models["lexicon_rule"])
    model_results["lexicon_rule"] = {
        "raw_label": lex_label,
        "mapped_label": lex_label,
        "score": float(lex_score),
    }
    votes[lex_label] += 1

    final_label = max(votes, key=votes.get)
    avg_conf = sum(v["score"] for v in model_results.values()) / len(model_results)

    return {
        "text": text,
        "final_label": final_label,
        "avg_confidence": avg_conf,
        "votes": votes,
        "by_model": model_results,
    }


def print_analysis(result):
    """Print a clean analysis summary."""
    print("Input:", result["text"])
    print("Final sentiment:", result["final_label"])
    print("Average confidence:", f"{result['avg_confidence']:.2%}")
    print("Votes:", result["votes"])
    print("Per-model output:", result["by_model"])
    print("-" * 70)


def run_examples(sentiment_models):
    """Run a few test examples for quick verification."""
    examples = [
        "I really enjoyed this product. It feels smooth and reliable.",
        "I feel disappointed by the delays and repeated failures in the system.",
        "The product works as expected; nothing exceptional, nothing terrible.",
    ]

    for text in examples:
        result = analyze_sentiment(text, sentiment_models)
        print_analysis(result)


def process_user_text(user_text, sentiment_models):
    """Analyze any user text and print sentiment details."""
    result = analyze_sentiment(user_text, sentiment_models)
    print_analysis(result)
    return result
